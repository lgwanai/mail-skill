   def cmd_classify(args: dict, db: MailDatabase, client: MailClient | None) -> dict:
       """Classify emails automatically or get classification for one email."""
       from mail_manager.classifier import EmailClassifier

       message_id = args.get("message_id")

       if message_id:
           # Classify single email
           email = db.get_email(message_id)
           if not email:
               return error_response("EMAIL_NOT_FOUND", f"Email {message_id} not found")

           classifier = EmailClassifier()
           classification = classifier.classify(email)

           # Save to database
           db.update_classification(
               message_id=message_id,
               importance=classification.importance,
               category=classification.category,
               confidence=classification.confidence
           )

           return success_response(
               message_id=message_id,
               importance=classification.importance,
               category=classification.category,
               confidence=classification.confidence,
               matched_rules=classification.matched_rules
           )
       else:
           # Classify all unclassified emails
           emails = db.search_emails(limit=1000)
           classifier = EmailClassifier()
           count = 0
           for email in emails:
               if email.get("importance") == "normal" and email.get("category") == "uncategorized":
                   classification = classifier.classify(email)
                   db.update_classification(
                       message_id=email["message_id"],
                       importance=classification.importance,
                       category=classification.category,
                       confidence=classification.confidence
                   )
                   count += 1
           return success_response(classified=count)
   ```

2. Add cmd_reclassify command:

   def cmd_reclassify(args: dict, db: MailDatabase, client: MailClient | None) -> dict:
       """Manually reclassify an email."""
       message_id = args.get("message_id")
       importance = args.get("importance")
       category = args.get("category")

       if not message_id:
           return error_response("MISSING_MESSAGE_ID", "message_id is required")

       email = db.get_email(message_id)
       if not email:
           return error_response("EMAIL_NOT_FOUND", f"Email {message_id} not found")

       # Update with manual override flag
       db.update_classification(
           message_id=message_id,
           importance=importance,
           category=category,
           confidence=1.0,  # Manual classification has full confidence
           manual_override=True
       )

       return success_response(
           message_id=message_id,
           importance=importance,
           category=category,
           manual_override=True
       )
   ```
