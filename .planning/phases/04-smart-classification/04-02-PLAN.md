1. Implement main classify method:

   ```python
   from collections import defaultdict

   def classify(self, email: dict) -> Classification:
       """Classify an email using configured rules."""
       matches = []
       for rule in self.rules:
           if self._match_rule(rule, email):
               matches.append(rule)

       if not matches:
           return Classification(
               importance="normal",
               category="uncategorized",
               confidence=0.5,
               matched_rules=[]
           )

       # Weighted voting for importance
       importance_scores: dict[str, float] = defaultdict(float)
       category_scores: dict[str, float] = defaultdict(float)
       total_weight = 0.0

       for rule in matches:
           if rule.importance:
               importance_scores[rule.importance] += rule.weight
           if rule.category:
               category_scores[rule.category] += rule.weight
           total_weight += rule.weight

       # Select highest scoring
       importance = max(importance_scores, key=importance_scores.get) if importance_scores else "normal"
       category = max(category_scores, key=category_scores.get) if category_scores else "uncategorized"

       # Confidence: weighted by match count and total possible weight
       max_possible_weight = sum(r.weight for r in self.rules) or 1.0
       confidence = min(1.0, total_weight / max_possible_weight)

       return Classification(
           importance=importance,
           category=category,
           confidence=confidence,
           matched_rules=[r.name for r in matches]
       )
   ```
