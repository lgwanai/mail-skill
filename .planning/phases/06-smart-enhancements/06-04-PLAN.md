```python
def get_enhanced_thread_timeline(
    db: MailDatabase,
    seed_message_id: str,
    include_sender_thread: bool = True,
    limit: int = 100,
) -> list[dict]:
    # Start with existing timeline
    timeline = db.get_thread_timeline(seed_message_id, limit)

    if not include_sender_thread or not timeline:
        return timeline

    # Get all unique participants
    participants = set()
    for email in timeline:
        if email.get("sender"):
            participants.add(email["sender"])
        if email.get("recipient"):
            for r in email["recipient"].split(","):
                participants.add(r.strip())

    # Find related emails by participants
    existing_ids = {e["message_id"] for e in timeline}
    for participant in participants:
        related = db.search_emails(sender=participant, limit=20)
        for email in related:
            if email["message_id"] not in existing_ids:
                timeline.append(email)
                existing_ids.add(email["message_id"])

    timeline.sort(key=lambda x: x.get("date") or "")
    return timeline
```

```python
from scripts.mail_manager.llm.client import LLMClient
from scripts.mail_manager.llm.prompts import THREAD_SUMMARY_PROMPT

def generate_thread_summary(
    llm_client: LLMClient,
    timeline: list[dict],
    current_email: dict,
) -> str:
    """Generate summary of email thread for context."""
    if len(timeline) <= 1:
        return ""

    thread_text = []
    for email in timeline:
        if email["message_id"] == current_email["message_id"]:
            continue
        thread_text.append(
            f"From: {email.get('sender', 'Unknown')}\n"
            f"Date: {email.get('date', 'Unknown')}\n"
            f"Subject: {email.get('subject', 'No Subject')}\n"
            f"Preview: {(email.get('body_text') or '')[:200]}...\n"
        )

    prompt = f"{THREAD_SUMMARY_PROMPT}\n\n{'---'.join(thread_text)}\n\nSummary:"
    response = llm_client.chat([{"role": "user", "content": prompt}], temperature=0.3, max_tokens=200)
    return response.content
```

```python
def format_thread_view(
    timeline: list[dict],
    current_message_id: str,
    llm_client: LLMClient | None = None,
) -> str:
    """Format thread timeline for display.

    Current email: Full detail (uses detail.py format_email_detail)
    Other emails: Summary line with sender, date, truncated subject
    """
    from scripts.mail_manager.detail import format_email_detail

    sections = ["## Thread Timeline"]

    current_email = None
    for email in timeline:
        if email["message_id"] == current_message_id:
            current_email = email
            break

    # Add thread summary if LLM available and multiple emails
    if llm_client and current_email and len(timeline) > 1:
        summary = generate_thread_summary(llm_client, timeline, current_email)
        if summary:
            sections.append(f"**Thread Summary:** {summary}")
            sections.append("")

    # Format each email
    for email in timeline:
        if email["message_id"] == current_message_id:
            sections.append("### Current Email")
            sections.append(format_email_detail(email))
        else:
            date = (email.get("date") or "")[:10]
            sender = email.get("sender", "Unknown")
            subject = (email.get("subject") or "No Subject")[:50]
            sections.append(f"- **{date}** {sender}: {subject}")

    return "\n".join(sections)
```
