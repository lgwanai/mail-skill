```python
def chat_with_history(
    self,
    system_prompt: str,
    conversation: list[dict[str, str]],
    user_message: str,
) -> LLMResponse:
    messages = [{"role": "system", "content": system_prompt}]
    messages.extend(conversation)
    messages.append({"role": "user", "content": user_message})
    return self.chat(messages)
```
