## Vision API Pattern

OpenAI Vision API uses base64-encoded images in message content:
```python
messages = [{
    "role": "user",
    "content": [
        {"type": "text", "text": "Describe this image"},
        {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{base64_image}"}}
    ]
}]
```

       """Save parsed content for an attachment."""
       with self._get_connection() as conn:
           cursor = conn.cursor()
           cursor.execute(
               "UPDATE attachments SET content_text = ? WHERE local_path = ?",
               (content, local_path)
           )
           conn.commit()

   def get_attachment_content(self, local_path: str) -> str | None:
       """Get parsed content for an attachment."""
       with self._get_connection() as conn:
           cursor = conn.cursor()
           cursor.execute(
               "SELECT content_text FROM attachments WHERE local_path = ?",
               (local_path,)
           )
           row = cursor.fetchone()
           return row["content_text"] if row else None

def parse_and_store_attachment(
    file_path: Path,
    db: MailDatabase,
    llm_client: LLMClient | None = None,

    """Parse attachment content and store in database.

    Returns parsed text content.
    """
    result = parse_attachment(file_path, llm_client)
    if result.text and file_path.exists():
        db.save_attachment_content(str(file_path), result.text)
